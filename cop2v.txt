--- 4. .github/workflows/build-ipa.yml — Full Content ---

name: BionicSX2 — Build IPA
on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]
  workflow_dispatch:

jobs:
  build-ios:
    name: Build iOS IPA
    runs-on: macos-14

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          submodules: recursive
          fetch-depth: 0

      - name: Select Xcode
        run: sudo xcode-select -s /Applications/Xcode_15.4.app

      - name: Install CMake
        run: brew install cmake ninja

      - name: Configure CMake for iOS
        run: |
          cmake -B build \
            -G Ninja \
            -DCMAKE_TOOLCHAIN_FILE=cmake/ios.toolchain.cmake \
            -DPCSX2_TARGET_IOS=ON \
            -DCMAKE_BUILD_TYPE=Release \
            -DDISABLE_PCSX2_RECOMPILER=ON \
            -DDISABLE_VULKAN=ON \
            -DDISABLE_ANDROID=ON \
            -D_M_ARM64=ON

      - name: Build
        run: cmake --build build --config Release --parallel

      - name: Archive IPA
        run: |
          xcodebuild archive \
            -scheme BionicSX2 \
            -sdk iphoneos \
            -archivePath build/BionicSX2.xcarchive \
            ONLY_ACTIVE_ARCH=NO \
            CODE_SIGN_IDENTITY="" \
            CODE_SIGNING_REQUIRED=NO \
            CODE_SIGNING_ALLOWED=NO

      - name: Export IPA
        run: |
          mkdir -p build/ipa
          cp -r build/BionicSX2.xcarchive/Products/Applications/BionicSX2.app \
            build/ipa/
          cd build/ipa
          mkdir Payload
          cp -r BionicSX2.app Payload/
          zip -r BionicSX2.ipa Payload/

      - name: Upload IPA artifact
        uses: actions/upload-artifact@v4
        with:
          name: BionicSX2-unsigned
          path: build/ipa/BionicSX2.ipa
          retention-days: 30
